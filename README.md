## Cool Poem
```rust
trait Red {
    fn is_red() -> bool {
        true
    }
}
trait Blue {
    fn is_blue() -> bool {
        true
    }
}
trait Win {
    fn is_win() -> bool {
        true
    }
}
trait You {
    fn is_you() -> bool {
        true
    }
}

struct Rose;
impl Red for Rose {}

struct Violet;
impl Blue for Violet {}

struct Flag;
impl Win for Flag {}

struct Baba;
impl You for Baba {}

fn main() {
    assert!(Rose::is_red());
    assert!(Violet::is_blue());
    assert!(Flag::is_win());
    assert!(Baba::is_you());
}
```
